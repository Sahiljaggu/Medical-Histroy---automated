    <form id="medicalForm">

        <label for="name">Name:</label>

        <input type="text" id="name" name="name" required><br><br>

       

        <label for="age">Age:</label>

        <input type="number" id="age" name="age" required><br><br>

 

        <label for="conditions">Medical Conditions:</label>

        <textarea id="conditions" name="conditions" placeholder="e.g., Diabetes, Hypertension"></textarea><br><br>

       

        <label for="medications">Current Medications:</label>

        <textarea id="medications" name="medications" placeholder="e.g., Metformin 500mg daily"></textarea><br><br>

 

        <label for="symptoms">Recent Symptoms:</label>

        <textarea id="symptoms" name="symptoms" placeholder="e.g., Chest pain, Fatigue"></textarea><br><br>

 

        <button type="button" onclick="saveData()">Save Medical History</button>

    </form>

 

    <h2>Medical History Summary</h2>

    <div id="summary"></div>

 

    <script>

        function saveData() {

            const name = document.getElementById("name").value;

            const age = document.getElementById("age").value;

            const conditions = document.getElementById("conditions").value;

            const medications = document.getElementById("medications").value;

            const symptoms = document.getElementById("symptoms").value;

 

            const summary = `

                <p><strong>Name:</strong> ${name}</p>

                <p><strong>Age:</strong> ${age}</p>

                <p><strong>Medical Conditions:</strong> ${conditions}</p>

                <p><strong>Current Medications:</strong> ${medications}</p>

                <p><strong>Recent Symptoms:</strong> ${symptoms}</p>

            `;

 

            document.getElementById("summary").innerHTML = summary;

            alert("Medical history saved!");

        }

    </script>
